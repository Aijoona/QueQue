QueQue :: simple queue

Simple sync queue

Code

// Basic queue
QueQue()
    .add(function(proxy) {
        return proxy.memo + ' mister';
    })
    .add(function(proxy) {
        return proxy.memo + ' world';
    })
    .add(function(proxy) {
        document.getElementById('res').innerHTML = proxy.memo + '!';
    })
    .start('Hello'); // Hello world!
