QueQue.batch(function(memo, curr, max) {
    if (curr == max) QueQue.end();
    return [memo + curr, curr + 1, max];
}, {
    args: [0, 1e7],
    cb: function(total) {
        document.getElementById('res').innerHTML = 'Sum: ' + total;
    }
});
